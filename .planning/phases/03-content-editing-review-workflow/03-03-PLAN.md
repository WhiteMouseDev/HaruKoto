**Step 2: Create `use-content-detail.ts`** (`apps/admin/src/hooks/use-content-detail.ts`)

```typescript
'use client';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';

export function useContentDetail<T>(contentType: string, id: string) {
  const queryClient = useQueryClient();

  const detailQuery = useQuery({
    queryKey: ['admin-content', contentType, id],
    queryFn: () => fetchAdminContentDetail<T>(contentType, id),
    staleTime: 30_000,
  });

  const patchMutation = useMutation({
    mutationFn: (data: Record<string, unknown>) => patchAdminContent(contentType, id, data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['admin-content', contentType, id] });
      queryClient.invalidateQueries({ queryKey: ['admin-content', contentType, 'list'] });
      // toast.success(t('edit.saveSuccess'))
    },
  });

  const reviewMutation = useMutation({
    mutationFn: ({ action, reason }: { action: 'approve' | 'reject'; reason?: string }) =>
      reviewContent(contentType, id, action, reason),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['admin-content', contentType, id] });
      queryClient.invalidateQueries({ queryKey: ['admin-content', contentType, 'list'] });
      queryClient.invalidateQueries({ queryKey: ['admin-content', 'stats'] });
    },
  });

  const auditQuery = useQuery({
    queryKey: ['admin-content', contentType, id, 'audit-logs'],
    queryFn: () => fetchAuditLogs(contentType, id),
    staleTime: 30_000,
  });

  return { detailQuery, patchMutation, reviewMutation, auditQuery };
}
```

**Step 3: Create `use-bulk-review.ts`** (`apps/admin/src/hooks/use-bulk-review.ts`)

```typescript
'use client';
export function useBulkReview(contentType: string) {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ ids, action, reason }: { ids: string[]; action: 'approve' | 'reject'; reason?: string }) =>
      batchReviewContent(contentType, ids, action, reason),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['admin-content', contentType] });
      queryClient.invalidateQueries({ queryKey: ['admin-content', 'stats'] });
    },
  });
}
```
