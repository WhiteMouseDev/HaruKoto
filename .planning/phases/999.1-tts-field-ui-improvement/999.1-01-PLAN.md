    Change from `handlePlayPause()` (no args) to `handlePlayPause(field: string)`:
    ```typescript
    function handlePlayPause(field: string) {
      const url = data?.audioUrl;
      if (!url) return;

      if (playingField === field) {
        // Same row — pause
        audioRef.current?.pause();
        setPlayingField(null);
      } else {
        // Different row or nothing playing — stop current, play new
        audioRef.current?.pause();
        if (!audioRef.current || audioRef.current.src !== url) {
          audioRef.current = new Audio(url);
          audioRef.current.addEventListener('ended', () => setPlayingField(null));
        } else {
          audioRef.current.currentTime = 0;
        }
        void audioRef.current.play();
        setPlayingField(field);
      }
    }
    ```

    **regenerateMutation refactor:**
    Change `mutationFn` to accept a `field: string` parameter:
    ```typescript
    const regenerateMutation = useMutation({
      mutationFn: (field: string) => regenerateTts(contentType, itemId, field),
      onSuccess: (newData) => {
        void queryClient.invalidateQueries({
          queryKey: ['admin-tts', contentType, itemId],
        });
        setConfirmField(null);

        // Auto-play new audio on the field that was just regenerated
        const regeneratedField = confirmField; // capture before reset
        if (newData.audioUrl) {
          audioRef.current?.pause();
          audioRef.current = new Audio(newData.audioUrl);
          audioRef.current.addEventListener('ended', () => setPlayingField(null));
          void audioRef.current.play();
          setPlayingField(regeneratedField);
        }

        toast.success('TTSを再生成しました');
      },
      onError: (err: Error) => {
        toast.error(err.message || '再生成に失敗しました。もう一度お試しください。');
        setConfirmField(null);
      },
    });
    ```

    NOTE: `confirmField` is captured before `setConfirmField(null)` in onSuccess because setState is async. Use a local variable to capture it OR move `setConfirmField(null)` after `setPlayingField`. The cleanest approach is to capture `confirmField` in a ref or use the mutation variables:
    ```typescript
    onSuccess: (newData, field) => {
      // 'field' is the variable passed to mutationFn
      ...
      setPlayingField(field);
      setConfirmField(null);
    },
    ```
    This is cleaner — `useMutation` passes the mutation variable as the second arg to `onSuccess`.

    **Return value changes:**

    ```tsx
    const fields = TTS_FIELDS[contentType].options;
    const hasAudio = !!audioUrl;

    return (
      <>
        <div className="rounded-lg border border-border bg-card">
          {fields.map((field, index) => (
            <div
              key={field.value}
              className={`flex items-center gap-2 p-3 ${
                index < fields.length - 1 ? 'border-b border-border' : ''
              }`}
            >
              {/* Status icon */}
              {hasAudio ? (
                <CheckCircle2 className="size-4 shrink-0 text-green-500" />
              ) : (
                <XCircle className="size-4 shrink-0 text-muted-foreground" />
              )}

              {/* Field name label */}
              <span className="flex-1 text-sm">
                {t(field.labelKey as Parameters<typeof t>[0])}
              </span>

              {/* Action buttons — audio present */}
              {hasAudio ? (
                <>
                  <Button
                    variant="outline"
                    size="icon"
                    onClick={() => handlePlayPause(field.value)}
                    aria-label={playingField === field.value ? t('pause') : t('play')}
                    className="size-8 shrink-0"
                  >
                    {playingField === field.value ? (
                      <Pause className="size-4" />
                    ) : (
                      <Play className="size-4" />
                    )}
                  </Button>
                  <Button
                    variant="ghost"
                    size="icon"
                    onClick={() => setConfirmField(field.value)}
                    aria-label={t('regenerateTooltip')}
                    className="size-8 shrink-0"
                  >
                    <RotateCcw className="size-4" />
                  </Button>
                </>
              ) : (
                <>
                  <span className="text-xs text-muted-foreground">
                    {t('noAudio')}
                  </span>
                  <Button
                    variant="default"
                    size="sm"
                    onClick={() => setConfirmField(field.value)}
                  >
                    {t('generate')}
                  </Button>
                </>
              )}
            </div>
          ))}
        </div>

        <RegenerateConfirmDialog
          open={confirmField !== null}
          onClose={() => setConfirmField(null)}
          onConfirm={() => {
            if (confirmField) {
              regenerateMutation.mutate(confirmField);
            }
          }}
          itemLabel={itemLabel}
          isLoading={regenerateMutation.isPending}
        />
      </>
    );
    ```
