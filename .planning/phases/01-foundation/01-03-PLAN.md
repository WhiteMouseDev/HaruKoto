    **src/app/api/locale/route.ts** (per RESEARCH Pattern 9):
    - Import NextResponse from 'next/server'
    - Import cookies from 'next/headers' (async -- Next.js 16)
    - export async function POST(request: Request):
      1. Parse body: const { locale } = await request.json()
      2. Validate: const validLocales = ['ja', 'ko', 'en']; if (!validLocales.includes(locale)) return 400
      3. const cookieStore = await cookies()
      4. cookieStore.set('NEXT_LOCALE', locale, { path: '/', maxAge: 365 * 24 * 60 * 60, sameSite: 'lax' })
      5. Return NextResponse.json({ locale })
