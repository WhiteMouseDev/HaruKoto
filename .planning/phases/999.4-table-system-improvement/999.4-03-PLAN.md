**B) Add sort handling to ContentTable:**

Import `useRouter`, `usePathname`, `useSearchParams` from `next/navigation` and `ArrowUp`, `ArrowDown`, `ArrowUpDown` from `lucide-react`.

Read current sort state from URL:
```tsx
const router = useRouter();
const pathname = usePathname();
const searchParams = useSearchParams();

const currentSortBy = searchParams.get('sort_by');
const currentSortOrder = searchParams.get('sort_order') ?? 'desc';
```

Add a sort toggle function:
```tsx
function handleSort(sortKey: string) {
  const params = new URLSearchParams(searchParams.toString());
  if (currentSortBy === sortKey) {
    // Toggle direction
    params.set('sort_order', currentSortOrder === 'asc' ? 'desc' : 'asc');
  } else {
    params.set('sort_by', sortKey);
    params.set('sort_order', 'desc'); // default desc when switching column
  }
  params.set('page', '1'); // reset to page 1 on sort change
  router.replace(pathname + '?' + params.toString());
}
```

In the TableHeader rendering, for columns with `sortKey`, render a clickable header:
```tsx
col.sortKey ? (
  <TableHead
    key={col.key}
    style={{ width: col.width }}
    className="cursor-pointer select-none text-xs uppercase tracking-wide text-muted-foreground hover:text-foreground"
    onClick={() => handleSort(col.sortKey!)}
  >
    <span className="inline-flex items-center gap-1">
      {col.header}
      {currentSortBy === col.sortKey ? (
        currentSortOrder === 'asc' ? (
          <ArrowUp className="size-3" />
        ) : (
          <ArrowDown className="size-3" />
        )
      ) : (
        <ArrowUpDown className="size-3 opacity-30" />
      )}
    </span>
  </TableHead>
) : (
  <TableHead
    key={col.key}
    style={{ width: col.width }}
    className="text-xs uppercase tracking-wide text-muted-foreground"
  >
    {col.header}
  </TableHead>
)
```
